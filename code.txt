#include<bits/stdc++.h>
using namespace std;

vector<int> digits(9000);

void calculate_digits()
{
    for(int k=1;k<=8999;k++)
    {
        digits[k]=0;
        for(int i=0;k*i<9000;i++)
        {
            cout<<k*i<<" ";
            digits[k]=digits[k]+log10(9000-k*i);
        }
        cout<<endl;
    }

}

int find_digit(int x,int n)
{
    if(x>31682)
        return 1;
    int l=0;
    int h=n;
 
    while(l<h)
    {
        int mid=(l+h)/2;
        
        if (x >= digits[mid]) 
        {
            l=mid+1;
            //h = mid;
        } 
        else 
        {
            h=mid;
            //l = mid + 1;
        }
    }
    return h;
    
}

int main()
{
    calculate_digits();
    cout<<digits[0]<<endl;
    cout<<digits[1]<<endl;
    cout<<digits[2]<<endl;
    cout<<digits[1990]<<endl;
    cout<<digits[156]<<endl;
    return 0;
}
